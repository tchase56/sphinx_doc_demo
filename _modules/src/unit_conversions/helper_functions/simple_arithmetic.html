    """

        a (float): The dividend

[docs]def floor_divide(a: float, b: float) -> int:
    """
    Perform floor division (integer division).
    
    Args:
        a (float): The dividend
        b (float): The divisor
    
    Returns:
        int: The floor quotient of a divided by b
    
    Raises:
        ValueError: If b is zero

    if b == 0:
